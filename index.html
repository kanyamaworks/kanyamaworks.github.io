Query 1 — GTM rep adoption rate by region
-- % of reps per region who ran the DE play this quarter SELECT r.region, COUNT(DISTINCT r.rep_id) AS total_reps, COUNT(DISTINCT CASE WHEN a.play_activated THEN r.rep_id END) AS active_reps, ROUND(active_reps / total_reps * 100, 1) AS adoption_pct FROM rep_master r LEFT JOIN activity_log a ON r.rep_id = a.rep_id AND a.play_type = 'DATA_ENGINEERING' AND a.quarter = 'Q2_FY2026' GROUP BY 1 ORDER BY adoption_pct DESC;
Query 2 — Whitespace accounts (core contract, zero DE attach)
SELECT a.account_name, a.vertical, a.account_tier, a.arr_usd AS current_arr, a.estimated_de_potential AS whitespace_arr, DATEDIFF('day', a.contract_start, CURRENT_DATE()) AS days_as_customer FROM account_master a WHERE a.snowflake_contract = TRUE AND a.account_id NOT IN ( SELECT DISTINCT account_id FROM product_usage WHERE product_family = 'DATA_ENGINEERING' AND usage_month >= DATEADD('month', -3, CURRENT_DATE()) ) ORDER BY whitespace_arr DESC LIMIT 50;
Query 3 — Product attach rate by tier, Q1 vs Q2
SELECT a.account_tier, p.product_name, COUNT(DISTINCT CASE WHEN p.quarter = 'Q1_FY2026' THEN a.account_id END) AS q1_accounts, COUNT(DISTINCT CASE WHEN p.quarter = 'Q2_FY2026' THEN a.account_id END) AS q2_accounts, COUNT(DISTINCT a.account_id) AS total_accounts, ROUND(q2_accounts / total_accounts * 100, 1) AS attach_pct_q2 FROM account_master a JOIN product_usage p ON a.account_id = p.account_id WHERE p.product_family = 'DATA_ENGINEERING' GROUP BY 1, 2 ORDER BY 1, attach_pct_q2 DESC;
